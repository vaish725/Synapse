│  🧠 GEMINI NANO QUICK SETUP - 15 MINUTES                    │
└─────────────────────────────────────────────────────────────┘

STEP 1: DOWNLOAD CHROME DEV/CANARY (5 min)
├─ Go to: https://www.google.com/chrome/dev/
├─ Download and install Chrome Dev
└─ Launch Chrome Dev (different from regular Chrome)

STEP 2: ENABLE FLAGS (2 min)
├─ Go to: chrome://flags
├─ Enable: "optimization-guide-on-device-model" 
│  └─ SET TO: "Enabled BypassPerfRequirement" ⚠️
├─ Enable: "prompt-api-for-gemini-nano" 
│  └─ SET TO: "Enabled"
└─ Click "Relaunch" button

STEP 3: DOWNLOAD MODEL (5-10 min)
├─ Go to: chrome://components
├─ Find: "Optimization Guide On Device Model"
├─ Click: "Check for update"
├─ Wait: ~1.5GB download (watch version number change)
└─ Verify: Version shows 2024.XX.XX.XXX (not 0.0.0.0)

STEP 4: VERIFY (1 min)
├─ Open console: Cmd + Option + J
├─ Paste: await window.ai.languageModel.capabilities()
└─ Expected: {available: "readily"}

STEP 5: TEST IN EXTENSION (2 min)
├─ Go to: chrome://extensions/
├─ Enable "Developer mode"
├─ Click "Load unpacked" → Select Synapse folder
├─ Click extension icon → Generate insights
└─ Should see: "Gemini Nano initialized successfully"

(async () => {
  const capabilities = await window.ai.languageModel.capabilities();
  console.log('Status:', capabilities);
  
  if (capabilities.available === 'readily') {
    const session = await window.ai.languageModel.create();
    const response = await session.prompt("Say hello in 5 words");
    console.log('AI says:', response);
    session.destroy();
  }
})();
